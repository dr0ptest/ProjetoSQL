/* Query 1 - os artistas que mais ganharam  */
SELECT 
	Artist.name Artist_Name,
	CAST(SUM(InvoiceLine.UnitPrice * InvoiceLine.Quantity) AS INT) Amount_Spent
FROM Invoice
	JOIN InvoiceLine
	ON Invoice.InvoiceId = InvoiceLine.InvoiceId
	JOIN Track
	ON InvoiceLine.TrackId = Track.TrackId
	JOIN Album
	ON Track.AlbumId = Album.AlbumId
	JOIN Artist
	ON Album.ArtistId = Artist.ArtistId
GROUP BY 1
ORDER BY 2 DESC
LIMIT 10;

/* Query 2 - artistas que mais escrevem música */

SELECT 
	Artist.Name Artist_name,
	COUNT(*) Songs
FROM Genre
	JOIN Track
	ON Genre.GenreId = Track.GenreId
	JOIN Album
	ON Album.AlbumId = Track.AlbumId
	JOIN Artist
	ON Artist.ArtistId = Album.ArtistId
WHERE Genre.Name = 'Rock'
GROUP BY 1
ORDER BY 2 DESC
LIMIT 10;

/* Query 3 - Generos que mais faturam */

SELECT 
	Genre.Name GenreName,
	CAST(SUM(Invoice.Total) AS INT) AmountSpent
FROM Genre
	JOIN Track
	ON Genre.GenreId = Track.GenreId
	JOIN InvoiceLine
	ON InvoiceLine.TrackId = Track.TrackId
	JOIN Invoice
	ON Invoice.InvoiceId = InvoiceLine.InvoiceId
GROUP BY 1
ORDER BY 2 DESC
LIMIT 10

/* Query 4 - Melhorer Clientes */

SELECT
	Customer.FirstName || ' ' || Customer.LastName AS full_name,
	CAST(SUM(Invoice.Total) AS INT) AS amountSpent 
FROM Invoice
JOIN Customer
ON Invoice.CustomerId = Customer.CustomerId
GROUP BY 1
HAVING amountSpent >= (
						SELECT CAST(AVG(sum_total) AS INT)
						FROM (
							SELECT 
								Invoice.CustomerId,
								SUM(Invoice.Total) sum_total
							FROM Invoice
							GROUP BY 1) sub)
ORDER BY 2 DESC
